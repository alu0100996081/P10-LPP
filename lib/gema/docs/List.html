def energia
    puntero=@head
    energia=0
    if(@head!=nil)
        while(puntero!=nil) do
            energia+=puntero.value.energia
            if(puntero.nest!=nil)
                puntero=puntero.nest
            else
                puntero=nil
            end
        end
    end
    return energia
end

def get_head
        if(@head == nil)
                puts "List Empty"
        else
                aux = @head
                @head = @head.nest
                if(head!=nil)
                        @head.prev = nil
                end
                aux.nest = nil
                if(@head == nil)
                        @tail = nil
                end
        end
        return aux
end

        nodo = Node.new(value,nil,nil)
        if(@head == nil)
                @tail = nodo
                @head = nodo
        else
                nodo.nest = @head
                @head.prev = nodo

                nodo.prev = nil
        end
end
